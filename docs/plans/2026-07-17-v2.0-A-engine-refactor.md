# v2.0-A 引擎重构实现计划(operator 化 + 线性光 + numba 融合)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把 `looklift/render.py` 的单函数管线重构为 operator 化、分阶段、线性光、numba 融合的渲染引擎,补齐 texture/clarity/dehaze/grain 像素实现并导出嵌 sRGB ICC,对外仍是 `render.render(image, analysis) -> Image`。

**Architecture:** 现 `_apply_color_ops`(9 步耦合大函数)+ `_apply_spatial_ops` 拆成 `looklift/render/` 包:每个调整是一个职责单一、可独测的 operator(`apply_numpy` 参考实现 + `apply_px` numba device 函数双形态);只有曝光/白平衡搬进线性光做物理正确运算,color grading 整体留在 display 段;pointwise 色彩 op 用单遍 `@njit(parallel,fastmath,cache)` 融合内核。texture/clarity/dehaze 在 S2 生成辅助缓冲后接入融合管线,grain 在 S2 生成噪声场、只在 S4 叠加一次。operator 是组织/契约/测试层,融合内核是执行层,共享同一份数学。

**Tech Stack:** Python 3.11+、numpy、Pillow、numba(BSD,新增)、pyvips(可选,ICC 嵌入)。

# looklift/render/contract.py
"""参数契约:可调参数点路径枚举 + 机器可读 (min,max) 范围 + hsl/global_ 解析规则。

v2.0-A 拥有的跨 spec 单一真相源(D1):v2.0-B 右面板 min/max/复位、v2.1 delta
白名单/clamp 域/落点都从本模块导出,不各自手抄。刻意只依赖 analyzer.ANALYSIS_SCHEMA,
不 import numba/pyvips —— chat 层导入它不触发引擎重编译。
"""
from __future__ import annotations

from ..analyzer import ANALYSIS_SCHEMA, _COLOR_KEYS

_BASIC_FIELDS = (
    "temperature_shift", "tint_shift", "exposure", "contrast",
    "highlights", "shadows", "whites", "blacks",
    "texture", "clarity", "dehaze", "vibrance", "saturation",
)
_HSL_FIELDS = ("hue", "saturation", "luminance")
_CG_ZONES = ("shadows", "midtones", "highlights", "global")  # 路径用 global(无下划线)
_CG_ZONE_FIELDS = ("hue", "saturation", "luminance")
_CG_SCALARS = ("blending", "balance")
_EFFECTS_FIELDS = ("vignette_amount", "grain_amount")


def param_paths() -> list[str]:
    """枚举所有可调数值参数的点路径。不含 tone_curve(整点数组非单值滑杆)。"""
    paths: list[str] = [f"basic.{f}" for f in _BASIC_FIELDS]
    for color in _COLOR_KEYS:
        paths += [f"hsl.{color}.{f}" for f in _HSL_FIELDS]
    for zone in _CG_ZONES:
        paths += [f"color_grading.{zone}.{f}" for f in _CG_ZONE_FIELDS]
    paths += [f"color_grading.{s}" for s in _CG_SCALARS]
    paths += [f"effects.{f}" for f in _EFFECTS_FIELDS]
    return paths


def param_bounds(path: str) -> tuple[float, float]:
    """从 schema 的 minimum/maximum 读该 path 范围。"""
    node = _schema_node(path)
    return (node["minimum"], node["maximum"])


def _schema_node(path: str) -> dict:
    props = ANALYSIS_SCHEMA["properties"]
    parts = path.split(".")
    section = parts[0]
    if section == "basic":
        return props["basic"]["properties"][parts[1]]
    if section == "effects":
        return props["effects"]["properties"][parts[1]]
    if section == "hsl":
        return props["hsl"]["items"]["properties"][parts[2]]
    if section == "color_grading":
        cg = props["color_grading"]["properties"]
        if parts[1] in _CG_SCALARS:
            return cg[parts[1]]
        key = "global_" if parts[1] == "global" else parts[1]
        return cg[key]["properties"][parts[2]]
    raise KeyError(path)


def resolve_path(analysis: dict, path: str) -> tuple[dict, str]:
    """把点路径解析到 analysis 落点,返回 (container, key),使 container[key] 是该值。

    封装两个结构怪癖:
      ① hsl 是数组,每项 {color, hue, saturation, luminance}:按 color 段 find,
         缺失则补零 {color, 0,0,0} 后 insert;
      ② color_grading.global 段映射到 dict 键 global_(避 Python 关键字)。
    """
    parts = path.split(".")
    section = parts[0]
    if section == "basic":
        return analysis.setdefault("basic", {}), parts[1]
    if section == "effects":
        return analysis.setdefault("effects", {}), parts[1]
    if section == "hsl":
        color, field = parts[1], parts[2]
        arr = analysis.setdefault("hsl", [])
        for item in arr:
            if item.get("color") == color:
                return item, field
        item = {"color": color, "hue": 0, "saturation": 0, "luminance": 0}
        arr.append(item)
        return item, field
    if section == "color_grading":
        cg = analysis.setdefault("color_grading", {})
        if parts[1] in _CG_SCALARS:
            return cg, parts[1]
        key = "global_" if parts[1] == "global" else parts[1]
        z = cg.setdefault(key, {"hue": 0, "saturation": 0, "luminance": 0})
        return z, parts[2]
    raise KeyError(path)
```

# looklift/render/base.py
"""Operator 协议、Stage/Domain、扁平 ResolvedParams 暂存 + 稳定 enable 位掩码。

单一数学、两处形态:apply_px(numba device 函数,被融合内核 inline,生产路径)与
apply_numpy(向量化参考,op 单测 / LUT 采样 / numba 兜底)。二者一致性由「融合前后
一致」测试(容差 ≤1/255)守护。operator 是组织/契约/测试层,融合内核是执行层。
"""
from __future__ import annotations

from enum import Enum, IntEnum
from typing import Protocol, runtime_checkable

import numpy as np


class Stage(IntEnum):
    """固定命名阶段序列(镜像 AlcedoStudio,非节点图)。"""
    INGEST = 0        # S0 PIL→float32、mode 归一、代理缩图
    TO_WORKING = 1    # S1 sRGB(display)→linear
    PREPASS = 2       # S2 辅助缓冲(模糊亮度/局部对比/噪声场/曲线烘 LUT)
    FUSED = 3         # S3 单遍融合色彩 op
    OUTPUT = 4        # S4 display→sRGB 输出、grain 叠加、嵌 ICC、编码回 PIL


class Domain(Enum):
    """op 执行的光域。物理合并运算在 LINEAR,LR 滑杆语义在 DISPLAY。"""
    LINEAR = "linear"
    DISPLAY = "display"


# 各 op 的 enable 位(位唯一;融合内核里 0 值 op 用 if 短路)。
# 顺序即 §2.3 单管线光域切换顺序:只有 exposure/WB 在 linear,其余在 display;
# grain 仅保留 S4 参数位,不得进入 fused 主体。
_OP_ORDER = (
    "exposure", "white_balance",                           # linear 段
    "contrast", "highlights_shadows", "whites_blacks",
    "tone_curve", "hsl", "saturation", "color_grading",
    "texture", "clarity", "dehaze", "vignette", "grain",   # display/S4 参数
)
OP_BITS: dict[str, int] = {name: 1 << i for i, name in enumerate(_OP_ORDER)}


@runtime_checkable
class Operator(Protocol):
    name: str
    stage: Stage
    domain: Domain

    def resolve(self, analysis: dict) -> tuple | None:
        """切出本 op 的参数分片,预计算成标量/小数组元组。全 0(无效)返回 None → enable=False。"""
        ...

    def apply_numpy(self, arr: np.ndarray, params: tuple, aux=None) -> np.ndarray:
        """整数组参考实现:op 单测 / LUT 采样 / numba 兜底。与 apply_px 是同一数学两种写法。"""
        ...

    def apply_px(
        self, r: float, g: float, b: float, *params, aux=None
    ) -> tuple[float, float, float]:
        """逐像素实现;T6 起由 numba 融合内核 inline。"""
        ...


def _contains_dict(value) -> bool:
    """递归拒绝 tuple/list 任意深度中的 dict。"""
    if isinstance(value, dict):
        return True
    if isinstance(value, (tuple, list)):
        return any(_contains_dict(item) for item in value)
    return False


class ResolvedParams:
    """operator resolve 结果的扁平 Python 暂存:op 名 → tuple + enable。

    tuple 只能由标量或定长小数组组成,禁止嵌套 dict。本结构供 numpy 参考路径使用,
    也是 T6b 固定 RenderParams 的输入;融合内核不得直接接收它。
    """

    __slots__ = ("_params", "enable")

    def __init__(self) -> None:
        self._params: dict[str, tuple] = {}
        self.enable: int = 0

    @classmethod
    def pack(cls, op_results: dict[str, tuple | None]) -> "ResolvedParams":
        rp = cls()
        for name, result in op_results.items():
            if result is None:
                continue                        # enable 位保持 0
            if not isinstance(result, tuple) or _contains_dict(result):
                raise TypeError("operator 参数必须是无嵌套 dict 的 tuple")
            rp._params[name] = result
            rp.enable |= OP_BITS[name]
        return rp

    def is_enabled(self, name: str) -> bool:
        return bool(self.enable & OP_BITS[name])

    def get(self, name: str) -> tuple | None:
        return self._params.get(name)
```

# looklift/render/color_space.py
"""sRGB↔linear 精确分段传递函数(纯 numpy)+ sRGB ICC 嵌入封装。

本期 sRGB-only:导出侧嵌标准 sRGB IEC61966-2.1 profile。Display-P3 基色转换
留位 v2.x(不在本期验收)。pyvips 可用时用 colourspace() 做 ICC-正确的 profile
写入,缺失时纯 numpy 传递函数 + Pillow 嵌 profile 兜底 —— 不让色管把引擎钉死在
libvips 安装成功上。
"""
from __future__ import annotations

from functools import lru_cache
from pathlib import Path

import numpy as np

try:  # pyvips 可选:仅用于 ICC-正确的 profile 写入,核心数学不依赖它
    import pyvips  # noqa: F401
    HAS_PYVIPS = True
except Exception:  # ImportError 或 libvips DLL 缺失
    HAS_PYVIPS = False

_ICC_PATH = Path(__file__).with_name("data") / "sRGB-IEC61966-2.1.icc"


def srgb_to_linear(arr: np.ndarray) -> np.ndarray:
    """sRGB(display, 0-1 编码)→ linear(scene-referred)。精确分段 EOTF(非 2.2 近似)。"""
    a = arr.astype(np.float32)
    return np.where(
        a <= 0.04045,
        a / 12.92,
        np.power((a + 0.055) / 1.055, 2.4),
    ).astype(np.float32)


def linear_to_srgb(arr: np.ndarray) -> np.ndarray:
    """linear → sRGB(display)。精确分段 OETF。仅本步前做范围收拢(linear 段不提前 clip)。"""
    a = arr.astype(np.float32)
    return np.where(
        a <= 0.0031308,
        a * 12.92,
        1.055 * np.power(np.clip(a, 0.0, None), 1.0 / 2.4) - 0.055,
    ).astype(np.float32)


@lru_cache(maxsize=1)
def srgb_icc_bytes() -> bytes:
    return _ICC_PATH.read_bytes()


def embed_icc(image) -> dict:
    """返回 Image.save 的 icc_profile kwargs(本期 sRGB)。pyvips 缺失时走 Pillow 原生嵌入。

    返回 dict 而非直接 save,便于调用方与其他 save 参数(quality/format)合并。
    """
    return {"icc_profile": srgb_icc_bytes()}
```

# looklift/render/operators/basic.py
"""basic 面板 op:exposure / white_balance / contrast / highlights_shadows / whites_blacks。

apply_numpy 的数学逐字搬自旧 _apply_color_ops(display 域,不改数值)。T5 起把
exposure/white_balance 迁入 linear;本任务仅纯搬家。
"""
from __future__ import annotations

import numpy as np

from ..base import Domain, Stage
from .._legacy import _luminance


class Exposure:
    name = "exposure"
    stage = Stage.FUSED
    domain = Domain.DISPLAY  # T5 迁 LINEAR

    def resolve(self, analysis):
        ev = analysis.get("basic", {}).get("exposure", 0)
        return None if not ev else (float(ev),)

    def apply_numpy(self, arr, params, aux=None):
        (ev,) = params
        return (arr.astype(np.float32) * (2.0 ** ev)).astype(np.float32)


class WhiteBalance:
    name = "white_balance"
    stage = Stage.FUSED
    domain = Domain.DISPLAY  # T5 迁 LINEAR

    def resolve(self, analysis):
        b = analysis.get("basic", {})
        t, tint = b.get("temperature_shift", 0), b.get("tint_shift", 0)
        return None if (not t and not tint) else (float(t), float(tint))

    def apply_numpy(self, arr, params, aux=None):
        t, tint = params
        arr = arr.astype(np.float32).copy()
        if t:
            arr[..., 0] *= 1 + 0.2 * t / 100
            arr[..., 2] *= 1 - 0.2 * t / 100
        if tint:
            arr[..., 1] *= 1 - 0.15 * tint / 100  # 品红=压绿
        return arr


class Contrast:
    name = "contrast"
    stage = Stage.FUSED
    domain = Domain.DISPLAY

    def resolve(self, analysis):
        c = analysis.get("basic", {}).get("contrast", 0)
        return None if not c else (float(c),)

    def apply_numpy(self, arr, params, aux=None):
        (c,) = params
        # 围绕 0.5 的线性扩张(±100 → ±60%)
        return (0.5 + (arr - 0.5) * (1 + 0.6 * c / 100)).astype(np.float32)


class HighlightsShadows:
    name = "highlights_shadows"
    stage = Stage.FUSED
    domain = Domain.DISPLAY

    def resolve(self, analysis):
        b = analysis.get("basic", {})
        hi, sh = b.get("highlights", 0), b.get("shadows", 0)
        return None if (not hi and not sh) else (float(hi), float(sh))

    def apply_numpy(self, arr, params, aux=None):
        hi, sh = params
        arr = np.clip(arr, 0, 1).astype(np.float32)
        luma = _luminance(arr)[..., None]
        if hi:
            arr = arr + (hi / 100) * 0.25 * (luma ** 2)       # 亮区权重 luma²
        if sh:
            arr = arr + (sh / 100) * 0.25 * ((1 - luma) ** 2)
        return arr.astype(np.float32)


class WhitesBlacks:
    name = "whites_blacks"
    stage = Stage.FUSED
    domain = Domain.DISPLAY

    def resolve(self, analysis):
        b = analysis.get("basic", {})
        wh, bl = b.get("whites", 0), b.get("blacks", 0)
        return None if (not wh and not bl) else (float(wh), float(bl))

    def apply_numpy(self, arr, params, aux=None):
        wh, bl = params
        arr = arr.astype(np.float32)
        if wh:
            arr = arr * (1 + 0.15 * wh / 100)
        if bl:
            arr = arr + 0.15 * bl / 100 * (1 - np.clip(arr, 0, 1)) ** 4  # 只影响近黑
        return arr.astype(np.float32)

# looklift/render/operators/tone.py
"""tone_curve op:0-255 控制点 → np.interp LUT,域外按斜率 1 外推(保证恒等)。

数学逐字搬自旧 _apply_color_ops 第 6 步(render.py:119-139)。curve 是小数组参数,
不进 contract.param_paths()。
"""
from __future__ import annotations

import numpy as np

from ..base import Domain, Stage


class ToneCurve:
    name = "tone_curve"
    stage = Stage.FUSED
    domain = Domain.DISPLAY

    def resolve(self, analysis):
        curve = sorted((p["input"], p["output"]) for p in analysis.get("tone_curve", []))
        if len(curve) < 2:
            return None
        xs = [p[0] for p in curve]
        ys = [p[1] for p in curve]
        # 域外按 45° 斜率外推补满 0-255,避免 np.interp 把域外像素夹平(GUI-T9 review)
        if xs[0] > 0:
            xs = [0] + xs
            ys = [ys[0] - xs[1]] + ys
        if xs[-1] < 255:
            ys = ys + [ys[-1] + (255 - xs[-1])]
            xs = xs + [255]
        xs = np.array(xs, dtype=np.float64) / 255.0
        ys = np.clip(np.array(ys, dtype=np.float64), 0, 255) / 255.0
        return (xs, ys)

    def apply_numpy(self, arr, params, aux=None):
        xs, ys = params
        return np.interp(arr, xs, ys).astype(np.float32)
```

```python
# looklift/render/operators/color.py
"""color op:hsl(8 通道定向)/ saturation(饱和+自然饱和)/ color_grading(亮度加权叠色)。

数学逐字搬自旧 _apply_color_ops 第 7/8/9 步(render.py:141-187)。HSV 基元迁移期复用
_legacy,T6 再内核化。HSL 八色键只从 analyzer._COLOR_KEYS 取得;color_grading 始终在 display。
"""
from __future__ import annotations

import numpy as np

from ..base import Domain, Stage
from ...analyzer import _COLOR_KEYS
from .._legacy import _rgb_to_hsv, _hsv_to_rgb, _luminance

_CENTER_VALUES = (0, 30, 60, 120, 180, 240, 280, 330)
_HSL_CENTERS = dict(zip(_COLOR_KEYS, _CENTER_VALUES, strict=True))


class Hsl:
    name = "hsl"
    stage = Stage.FUSED
    domain = Domain.DISPLAY

    def resolve(self, analysis):
        entries = [e for e in analysis.get("hsl", [])
                   if _HSL_CENTERS.get(e.get("color", "")) is not None
                   and (e.get("hue", 0) or e.get("saturation", 0) or e.get("luminance", 0))]
        return None if not entries else (entries,)

    def apply_numpy(self, arr, params, aux=None):
        (entries,) = params
        hsv = _rgb_to_hsv(arr)
        for entry in entries:
            center = _HSL_CENTERS[entry["color"]]
            dist = np.abs((hsv[..., 0] - center + 180) % 360 - 180)
            mask = np.clip(1 - dist / 45, 0, 1)  # 中心±45° 三角权重
            hsv[..., 0] = (hsv[..., 0] + mask * entry.get("hue", 0) * 0.3) % 360
            hsv[..., 1] *= 1 + mask * entry.get("saturation", 0) / 100 * 0.5
            hsv[..., 2] *= 1 + mask * entry.get("luminance", 0) / 100 * 0.3
        hsv[..., 1] = np.clip(hsv[..., 1], 0, 1)
        return _hsv_to_rgb(hsv)


class Saturation:
    name = "saturation"
    stage = Stage.FUSED
    domain = Domain.DISPLAY

    def resolve(self, analysis):
        b = analysis.get("basic", {})
        sat, vib = b.get("saturation", 0), b.get("vibrance", 0)
        return None if (not sat and not vib) else (float(sat), float(vib))

    def apply_numpy(self, arr, params, aux=None):
        sat, vib = params
        hsv = _rgb_to_hsv(arr)
        if sat:
            hsv[..., 1] *= 1 + sat / 100
        if vib:  # 自然饱和度:低饱和像素受益更多
            hsv[..., 1] += (vib / 100) * 0.5 * hsv[..., 1] * (1 - hsv[..., 1])
        hsv[..., 1] = np.clip(hsv[..., 1], 0, 1)
        return _hsv_to_rgb(hsv)


class ColorGrading:
    name = "color_grading"
    stage = Stage.FUSED
    domain = Domain.DISPLAY  # 艺术染色整体留在 display;blending/balance contract-only

    def resolve(self, analysis):
        cg = analysis.get("color_grading", {})
        active = any(cg.get(z, {}).get("saturation", 0) or cg.get(z, {}).get("luminance", 0)
                     for z in ("shadows", "midtones", "highlights", "global_"))
        return None if not active else (cg,)

    def apply_numpy(self, arr, params, aux=None):
        (cg,) = params
        luma = _luminance(arr)[..., None]
        weights = {
            "shadows": (1 - luma) ** 2, "midtones": 4 * luma * (1 - luma),
            "highlights": luma ** 2, "global_": np.ones_like(luma),
        }
        for zone, w in weights.items():
            z = cg.get(zone, {})
            s = z.get("saturation", 0)
            lum = z.get("luminance", 0)
            if not s and not lum:
                continue
            if s:
                hue = np.deg2rad(z.get("hue", 0))
                tint = (np.array(
                    [np.cos(hue), np.cos(hue - 2.0944), np.cos(hue - 4.1888)],
                    dtype=np.float32) * 0.5 + 0.5)
                arr = arr + w * (s / 100) * 0.3 * (tint - arr)
            if lum:
                arr = arr + w * lum / 100 * 0.2
        return arr.astype(np.float32)
```

# looklift/render/pipeline.py  (本任务新增 resolve_params/apply_color_ops_numpy)
"""分阶段编排 S0-S4;resolve_params(analysis)->ResolvedParams;proxy/full 两入口。

本任务先落地 display 域 numpy 串联(apply_color_ops_numpy),数值等价旧
_apply_color_ops,作为 numba 兜底路径与 LUT 采样入口的基座。T5 起插入 linear
编解码,T6 起融合内核成为生产路径、本串联退为兜底/参考。
"""
from __future__ import annotations

import numpy as np

from .base import ResolvedParams
from .operators import REGISTRY


def resolve_params(analysis: dict) -> ResolvedParams:
    """遍历注册表解析一次,生成 numpy 路径与 T6b marshal 共用的扁平暂存。"""
    return ResolvedParams.pack({op.name: op.resolve(analysis) for op in REGISTRY})


def apply_color_ops_numpy(arr: np.ndarray, analysis: dict) -> np.ndarray:
    """display 域 operator 串联参考实现。数值等价 _legacy._apply_color_ops。

    clip 点按旧顺序显式落在串联层(highlights_shadows 前后各一次、末尾一次),
    与 legacy 逐像素一致;单 op 的 apply_numpy 不自带这些串联级 clip。
    """
    arr = arr.astype(np.float32)
    ops = {op.name: op for op in REGISTRY}
    resolved = resolve_params(analysis)

    for name in ("exposure", "white_balance", "contrast"):
        op = ops[name]
        p = resolved.get(name)
        if p is not None:
            arr = op.apply_numpy(arr, p)

    arr = np.clip(arr, 0, 1)                      # legacy render.py:102
    op = ops["highlights_shadows"]
    p = resolved.get("highlights_shadows")
    if p is not None:
        arr = op.apply_numpy(arr, p)

    op = ops["whites_blacks"]
    p = resolved.get("whites_blacks")
    if p is not None:
        arr = op.apply_numpy(arr, p)
    arr = np.clip(arr, 0, 1)                      # legacy render.py:117

    for name in ("tone_curve", "hsl", "saturation", "color_grading"):
        op = ops[name]
        p = resolved.get(name)
        if p is not None:
            arr = op.apply_numpy(arr, p)

    return np.clip(arr, 0, 1).astype(np.float32)  # legacy render.py:163

from .operators import REGISTRY


def render_arr(arr_srgb: np.ndarray, analysis: dict) -> np.ndarray:
    """S1 linear 段(exposure/WB)→ 编码 display → display 段(含 color_grading)。

    §2.3 单管线内只切换一次光域;§2.4 linear 段内不提前 clip(中间值可 >1)。

    arr = cs.srgb_to_linear(arr_srgb.astype(np.float32))   # S1 display→linear

    # ---- linear 段:物理合并运算,不 clip ----
    for name in ("exposure", "white_balance"):
        op = ops[name]
        p = op.resolve(analysis)
        if p is not None:
            arr = op.apply_numpy(arr, p)
    arr = cs.linear_to_srgb(arr)                            # 唯一一次光域切换

    # ---- display 段:LR 滑杆语义 ----
    arr = np.clip(arr, 0, 1)

    if p is not None:
        arr = op.apply_numpy(arr, p)
    arr = np.clip(arr, 0, 1)
    op = ops["highlights_shadows"]; p = op.resolve(analysis)
    if p is not None:
        arr = op.apply_numpy(arr, p)
    op = ops["whites_blacks"]; p = op.resolve(analysis)
    if p is not None:
        arr = op.apply_numpy(arr, p)
    arr = np.clip(arr, 0, 1)
    for name in ("tone_curve", "hsl", "saturation", "color_grading"):
        op = ops[name]; p = op.resolve(analysis)

    return np.asarray(image, dtype=np.float32) / 255.0


def _to_image(arr):

def render_full(image, analysis):

def _render_to_pil(image, analysis):
    """执行已定义的 S0→S2→fused→S4 路径,返回尚未注入 ICC 的 PIL Image。"""
    src = _to_arr(image if image.mode == "RGB" else image.convert("RGB"))
    resolved = resolve_params(analysis)
    aux = prepare_aux(src, analysis)
    arr = render_fused(src, analysis, aux=aux)
    arr = apply_output_grain(arr, resolved, aux.noise)  # 唯一 grain 叠加
    return _to_image(arr)


def render(image, analysis):
    """生产渲染入口:返回的 PIL Image 自身即携带默认 sRGB ICC。"""
    from . import color_space as cs
    pil = _render_to_pil(image, analysis)
    pil.info["icc_profile"] = cs.srgb_icc_bytes()
    return pil


def export(image, analysis, out_path):
    """S0-S4 全跑,输出带 sRGB ICC 的成片。本期 sRGB-only(P3 留 v2.x)。"""
    pil = render(image, analysis)  # 与默认公开入口严格同一路径
    pil.save(out_path, icc_profile=pil.info["icc_profile"])
    return out_path

# tests/test_lut_render_consistency.py
import numpy as np
from PIL import Image
from looklift import lut
from looklift.render import pipeline


def _cube_lookup(path, rgb):
    """解析本项目 .cube(R 最快),对任意 RGB 做可执行的三线性插值。"""
    lines = [line.strip() for line in path.read_text(encoding="ascii").splitlines()]
    size = int(next(line.split()[1] for line in lines if line.startswith("LUT_3D_SIZE")))
    rows = [line for line in lines if line and line[0] not in '#ABCDEFGHIJKLMNOPQRSTUVWXYZ']
    table = np.asarray([[float(x) for x in line.split()] for line in rows], np.float32)
    table = table.reshape(size, size, size, 3)  # [b, g, r],R 变化最快
    pos = np.clip(np.asarray(rgb, np.float32), 0, 1) * (size - 1)
    lo = np.floor(pos).astype(int); hi = np.minimum(lo + 1, size - 1); f = pos - lo
    out = np.zeros(3, np.float32)
    for db in (0, 1):
        for dg in (0, 1):
            for dr in (0, 1):
                idx = (hi[2] if db else lo[2], hi[1] if dg else lo[1],
                       hi[0] if dr else lo[0])
                w = ((f[2] if db else 1-f[2]) * (f[1] if dg else 1-f[1]) *
                     (f[0] if dr else 1-f[0]))
                out += w * table[idx]
    return out


def _a():
    return {"basic": {"temperature_shift": 30, "tint_shift": 0, "exposure": 0.6,
            "contrast": 20, "highlights": 0, "shadows": 0, "whites": 0, "blacks": 0,
            "texture": 0, "clarity": 0, "dehaze": 0, "vibrance": 0, "saturation": 10},
            "tone_curve": [], "hsl": [],
            "color_grading": {z: {"hue": 0, "saturation": 0, "luminance": 0}
                              for z in ("shadows", "midtones", "highlights", "global_")}
            | {"blending": 50, "balance": 0},
            "effects": {"vignette_amount": 0, "grain_amount": 0}}


def test_cube_output_matches_render_at_sample_points(tmp_path):
    """D6:.cube 采样点走完整光域往返,与 render 对应像素输出一致(容差内)。"""

    # 解析 cube 网格,取若干采样 RGB,断言 ≈ render_arr 对同色输出
    for rgb in ([0.25, 0.25, 0.25], [0.5, 0.4, 0.3], [0.8, 0.2, 0.2]):
        src = np.array(rgb, dtype=np.float32).reshape(1, 1, 3)
        expected = pipeline.render_arr(src.copy(), _a())[0, 0]
        got = _cube_lookup(out, rgb)            # 三线性查表(测试辅助)
